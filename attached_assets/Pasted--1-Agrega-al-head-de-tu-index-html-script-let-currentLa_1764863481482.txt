<script>
let currentLang = 'es'; // Espa침ol por defecto

async function initTranslation() {
  // Detecta idioma del navegador
  const browserLang = navigator.language.split('-')[0];
  currentLang = ['es','en','pt'].includes(browserLang) ? browserLang : 'es';
  
  // Traduce todo autom치ticamente
  await translateAll(currentLang);
}

async function translateAll(lang) {
  const elements = document.querySelectorAll('[data-translate]');
  for(let el of elements) {
    try {
      const response = await fetch('https://libretranslate.de/translate', {
        method: 'POST',
        body: JSON.stringify({
          q: el.dataset.original || el.textContent,
          source: 'es',
          target: lang,
          format: 'text'
        }),
        headers: { 'Content-Type': 'application/json' }
      });
      const data = await response.json();
      el.textContent = data.translatedText;
    } catch(e) {
      console.log('Traducci칩n fall칩:', e);
    }
  }
  currentLang = lang;
}

// Botones de cambio de idioma
function switchTo(lang) {
  translateAll(lang);
  localStorage.setItem('lang', lang);
}
</script>
